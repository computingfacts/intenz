SOURCE VARCHAR2(6 BYTE) No null 6 Database internal code, for example 'INTENZ', 'IUBMB', 'BRENDA', 'SIB' etc. Each enzyme belongs to one of them.
NOTE VARCHAR2(200 BYTE) Yes null 7 Ambiguous, misleading, obsolete names have sometimes a comment attached to them which can be stored in this column.
ORDER_IN NUMBER(4,0) No null 8 Order applied to names for the same enzyme.
WEB_VIEW VARCHAR2(50 BYTE) No null 9 View to display the name in.
